_________________________________
ACQUIRING DATA
_________________________________

Running the following commands on vhh_core can be used to generate the data.

# Manual annotations
python download_annotation_results.py -p /data/ext/VHH/datasets/HistShotDS_V2/annotations/manual  -s NARA -a --manual --json
python download_annotation_results.py -p /data/ext/VHH/datasets/HistShotDS_V2/annotations/manual  -s EFA -a --manual --json
python download_annotation_results.py -p /data/ext/VHH/datasets/HistShotDS_V2/annotations/manual  -s LOC -a --manual --json

# Automatic annotations
python download_annotation_results.py -p /data/ext/VHH/datasets/HistShotDS_V2/annotations/automatic  -s NARA -a --json
python download_annotation_results.py -p /data/ext/VHH/datasets/HistShotDS_V2/annotations/automatic  -s EFA -a --json
python download_annotation_results.py -p /data/ext/VHH/datasets/HistShotDS_V2/annotations/automatic  -s LOC -a --json

# Metadata
python download_video_metadata.py -p /data/ext/VHH/datasets/HistShotDS_V2 -s NARA LOC EFA -r processed_shots processed_objects processed_relations processed_overscan

# Videos

python download_videos.py -s NARA -n -a -p /data/ext/VHH/datasets/HistShotDS_V2/films --full_name
python download_videos.py -s EFA -n -a -p /data/ext/VHH/datasets/HistShotDS_V2/films --full_name
python download_videos.py -s LOC -n -a -p /data/ext/VHH/datasets/HistShotDS_V2/films --full_name
